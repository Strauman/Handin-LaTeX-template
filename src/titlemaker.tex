\makeatletter
\let\old@maketitle = \maketitle
\gdef\maketitle{
	\old@maketitle
	\thispagestyle{empty}
	\clearpage
}
\gdef\@maketitle{%

	\begin{center}
		\ifshowoff
			\Huge{\textbackslash logo}\\
			\Large{\@title} \\[1.5cm]
		\else
		\ifset@logo{
			\includegraphics[width=0.5\textwidth]{\@logo}\\[0.5cm]
			\Large{\@title} \\[1.5cm]
		}{
			\ea\@latex@warning{Use \noexpand\logo{path/to/image} to set a logo on the front page}
			\Huge{\@title} \\[1.0cm]
		}
		\fi
	\end{center}
	\begin{center}
		\textbf{\@coursename~--~\@coursetitle}\\[1cm]
		\textbf{\@author} \\[1cm]
		\textbf{\today} \\[3cm]
	\end{center}

	\begin{center}
		\vfill
		\ifset@containspages{
			\@containspages
		}{
			\@tr{containspages}
		}\\[0.5cm]
		\textsc{\@institute}
	\end{center}
	\rhead{\@author}
	\lhead{\@coursename - \@coursetitle}
	\ifset@pagetext{
		\rfoot{\@pagetext}
	}{
		\rfoot{\@tr{pagetext}}
	}
}
\makeatother
